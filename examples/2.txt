Private Sub CommandButton2_Click()
    Dim strFileToimport As String, i As Long, j As Long
    Dim targetrange As Range, wb As Workbook, user As String
    Dim today As Date
    
    jcl = 2
    user = Application.UserName
    Set wb = ThisWorkbook
    Set targetrange = wb.Sheets("File_Updates").Cells(jcl, 1)
    today = Now
    While (Len(targetrange.Cells(jcl, 1).Value) <> 0)
        strFileToimport = targetrange.Cells(jcl, 1).Value
        targetrange.Cells(jcl, 4).Value = today
        targetrange.Cells(jcl, 5).Value = user
        
        'the previous info sets who was using the trending code and on what day, and which file has been run
        'in the files updates tab in excel workbook
        
        finish_info_fill (strFileToimport)
        jcl = jcl + 1
    Wend
End Sub

Private Sub CommandButton1_Click() 'this code is to find files to compare to the trendline graphs
    Dim strFileToimport As String, i As Long, j As Long
    Dim targetrange As Range, wb As Workbook, user As String
    Dim today As Date
    
    strFileToimport = Application.GetOpenFilename _
    (Title:="Please choose a file to import")
    
    If strFileToimport = "False" Then
        MsgBox "No file selected.", vbExclamation, "Sorry!" 'If chose a file that has already been run
        Exit Sub                                           'error message will appear
    Else
        i = 1
        user = Application.UserName
        Set wb = ThisWorkbook
        Set targetrange = wb.Sheets("File Updates").UsedRange
        today = Now
        
        While (Len(targetrange.Cells(i, 1).Value) <> 0)
            i = i + 1
        Wend
        
        For j = 1 To i
            If (targetrange(j, 1).Value = strFileToimport) Then
                MsgBox "This engine file has already been entered."
                Exit Sub
            End If
        Next j
        
        targetrange.Cells(i, 1).Value = strFileToimport
        targetrange.Cells(i, 4).Value = today
        targetrange.Cells(i, 5) = user
        
        'the previous info sets who was using the trending code and on what day, and which file has been run
        'in the files updates tab in excel workbook
        
        finish_info_fill (strFileToimport)
    End If
End Sub

Sub DtSrt()
    Dim StRw As Integer, EndRw As Integer
    Sheets("Take_Off").Select
    StRw = 2 ' Starting Row
    EndRw = ActiveSheet.Range("B65500").End(xlUp).Row
    ActiveSheet.Rows(StRw & ":" & EndRw).Select
    Selection.Sort Key1:=ActiveSheet.Range("C2"), Order1:=xlDescending
End Sub
